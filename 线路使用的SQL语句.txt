SELECT LINEID, NUM, LINENAME, LINETYPE, LINECLASS, REMARK, ASTATION, BSTATION, 
      MILES, CHANQUAN, CHANQUANGUISHOU
FROM TRAINLINE
WHERE (NOT EXISTS
          (SELECT ID, LINEID, NUM, ASTATION, BSTATION, MILES, MILESCLASS, 
               DIRECTION
         FROM LINESTATION
         WHERE (LINEID = TRAINLINE.LINEID)))
